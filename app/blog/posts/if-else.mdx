Aqui le mostramos como se trabajan los IF ELSE en Pascal.

```pascal
program IfElseExample;

var
  num: Integer;

begin
  write('Ingrese un número: ');
  readln(num);

  if num > 0 then
    writeln('El número es positivo')
  else if num < 0 then
    writeln('El número es negativo')
  else
    writeln('El número es cero');
end.
```
